时间复杂度O（N）
```c++
bool isprime[10000005];
int prime[1000004];
int cnt = 0;

void sift(int n)
{
    memset(isprime,true,sizeof(isprime));
    for(int i=2; i<=n; i++){

        if(isprime[i])
            prime[cnt++] = i;
        for(int j=0; j<cnt&&prime[j]*i<=n; j++){
            isprime[prime[j]*i] = false;
            if(i%prime[j]==0) break;
        }
    }
}
```
